## 🎯 实战项目：智能旅游助手

### 项目简介

构建一个智能旅游助手，功能包括：
1. 查询城市天气
2. 推荐旅游景点
3. 计算旅行预算
4. 生成旅游计划

import { tool } from '@langchain/core/tools';
import { z } from 'zod';

export const weatherTool = tool(
  async ({ city }) => {
    // 实际项目应调用真实天气 API
    const mockWeather: Record<string, any> = {
      '北京': { temp: 25, condition: '晴天', suggestion: '适合出游' },
      '上海': { temp: 28, condition: '多云', suggestion: '建议带伞' },
      '成都': { temp: 22, condition: '阴天', suggestion: '适合户外活动' },
    };

    const weather = mockWeather[city] || { temp: 20, condition: '未知', suggestion: '暂无数据' };
    return JSON.stringify(weather);
  },
  {
    name: 'get_weather',
    description: '获取指定城市的天气信息，包括温度、天气状况和出行建议',
    schema: z.object({
      city: z.string().describe('城市名称，如：北京、上海'),
    }),
  }
);

**`src/tools/budget.tool.ts`**

```typescript
import { tool } from '@langchain/core/tools';
import { z } from 'zod';

export const budgetTool = tool(
  async ({ city, days, level }) => {
    const dailyBudget: Record<string, Record<string, number>> = {
      '北京': { '经济': 300, '舒适': 600, '豪华': 1200 },
      '上海': { '经济': 400, '舒适': 800, '豪华': 1500 },
      '成都': { '经济': 200, '舒适': 400, '豪华': 800 },
    };

    const daily = dailyBudget[city]?.[level] || 500;
    const total = daily * days;

    return JSON.stringify({
      city,
      days,
      level,
      dailyBudget: daily,
      totalBudget: total,
      breakdown: {
        accommodation: daily * 0.4,
        food: daily * 0.3,
        transportation: daily * 0.2,
        attractions: daily * 0.1,
      },
    });
  },
  {
    name: 'calculate_budget',
    description: '计算旅行预算，包括住宿、餐饮、交通和门票',
    schema: z.object({
      city: z.string().describe('城市名称'),
      days: z.number().describe('旅行天数'),
      level: z.enum(['经济', '舒适', '豪华']).describe('消费水平'),
    }),
  }
);
```

### Step 3: 定义数据结构

**`src/schemas/travel-plan.schema.ts`**

```typescript
import { z } from 'zod';

export const travelPlanSchema = z.object({
  destination: z.string().describe('目的地城市'),
  duration: z.number().describe('旅行天数'),
  budget: z.object({
    total: z.number(),
    daily: z.number(),
  }),
  itinerary: z.array(
    z.object({
      day: z.number(),
      activities: z.array(z.string()),
      meals: z.array(z.string()),
    })
  ).describe('每日行程安排'),
  tips: z.array(z.string()).describe('旅行建议'),
});

export type TravelPlan = z.infer<typeof travelPlanSchema>;
```

### Step 4: 创建助手类

**`src/assistant.ts`**

```typescript
import { ChatOpenAI } from '@langchain/openai';
import { HumanMessage, AIMessage, ToolMessage, BaseMessage } from '@langchain/core/messages';
import { weatherTool, attractionsTool, budgetTool } from './tools';
import { travelPlanSchema, TravelPlan } from './schemas/travel-plan.schema';

export class TouristAssistant {
  private llm: ChatOpenAI;
  private tools: any[];

  constructor() {
    this.tools = [weatherTool, attractionsTool, budgetTool];

    this.llm = new ChatOpenAI({
      modelName: 'gpt-4o-mini',
      temperature: 0.7,
    }).bindTools(this.tools);
  }

  /**
   * 普通对话模式
   */
  async chat(userInput: string): Promise<string> {
    const messages: BaseMessage[] = [new HumanMessage(userInput)];

    while (true) {
      const response = await this.llm.invoke(messages);
      messages.push(response);

      if (!response.tool_calls || response.tool_calls.length === 0) {
        return response.content as string;
      }

      // 执行所有工具调用
      for (const toolCall of response.tool_calls) {
        console.log(`🔧 调用工具: ${toolCall.name}`);

        const tool = this.tools.find((t) => t.name === toolCall.name);
        if (tool) {
          const result = await tool.invoke(toolCall.args);
          console.log(`📤 结果: ${result}`);

          messages.push(
            new ToolMessage({
              content: result,
              tool_call_id: toolCall.id!,
            })
          );
        }
      }
    }
  }

  /**
   * 生成结构化旅游计划
   */
  async generatePlan(userInput: string): Promise<TravelPlan> {
    const llmWithSchema = new ChatOpenAI({
      modelName: 'gpt-4o-mini',
      temperature: 0.7,
    }).withStructuredOutput(travelPlanSchema);

    return await llmWithSchema.invoke(userInput);
  }
}
```

### Step 5: 主程序

**`src/index.ts`**

```typescript
import { TouristAssistant } from './assistant';
import 'dotenv/config';

async function main() {
  const assistant = new TouristAssistant();

  console.log('🌏 智能旅游助手已启动\n');

  // 测试 1：查询天气
  console.log('='.repeat(60));
  console.log('📍 场景 1：查询北京天气');
  console.log('='.repeat(60));
  const weather = await assistant.chat('北京现在天气怎么样？');
  console.log('💬', weather, '\n');

  // 测试 2：景点推荐
  console.log('='.repeat(60));
  console.log('📍 场景 2：推荐上海的历史景点');
  console.log('='.repeat(60));
  const attractions = await assistant.chat('推荐上海的历史景点');
  console.log('💬', attractions, '\n');

  // 测试 3：预算计算
  console.log('='.repeat(60));
  console.log('📍 场景 3：计算成都3天舒适游预算');
  console.log('='.repeat(60));
  const budget = await assistant.chat('帮我算一下成都3天舒适游的预算');
  console.log('💬', budget, '\n');

  // 测试 4：生成旅游计划
  console.log('='.repeat(60));
  console.log('📍 场景 4：生成北京3日游计划');
  console.log('='.repeat(60));
  const plan = await assistant.generatePlan(`
    帮我制定一个北京3日游计划：
    - 预算：每天600元
    - 喜欢历史文化景点
    - 想尝试北京特色美食
  `);
  console.log('📋 旅游计划：');
  console.log(JSON.stringify(plan, null, 2));
}

main();
```
