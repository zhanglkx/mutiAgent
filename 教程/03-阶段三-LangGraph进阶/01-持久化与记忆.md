# 01 - 持久化与记忆

> **核心理念**：让 Agent 拥有"记忆"，记住之前的对话和状态

## 🎯 学习目标

- 理解为什么需要持久化
- 掌握 Checkpointer 的使用
- 学会管理多用户对话
- 实现长期记忆功能

## 📝 为什么需要持久化？

### 问题：Agent 没有记忆

```typescript
// 第一次对话
await agent.invoke({ messages: [new HumanMessage('我叫小明')] });
// AI: "你好，小明！"

// 第二次对话（新的调用）
await agent.invoke({ messages: [new HumanMessage('我叫什么名字？')] });
// AI: "我不知道你的名字"  ❌ 忘记了！
```

**解决方案**：Checkpointer（检查点）

```typescript
import { MemorySaver } from '@langchain/langgraph';

const memory = new MemorySaver();
const agent = createReactAgent({ llm, tools, checkpointer: memory });

// 第一次对话（指定 thread_id）
await agent.invoke(
  { messages: [new HumanMessage('我叫小明')] },
  { configurable: { thread_id: 'user-123' } }
);

// 第二次对话（相同 thread_id，自动加载历史！）
await agent.invoke(
  { messages: [new HumanMessage('我叫什么名字？')] },
  { configurable: { thread_id: 'user-123' } }
);
// AI: "你叫小明"  ✅ 记住了！
```

## 🛠️ Checkpointer 基础

### 1. MemorySaver（内存存储）

```typescript
import { MemorySaver } from '@langchain/langgraph';
import { createReactAgent } from '@langchain/langgraph/prebuilt';
import { ChatOpenAI } from '@langchain/openai';
import { HumanMessage } from '@langchain/core/messages';

// 创建内存存储
const memory = new MemorySaver();

// 创建带记忆的 Agent
const agent = createReactAgent({
  llm: new ChatOpenAI({ modelName: 'gpt-4o-mini' }),
  tools: [],
  checkpointer: memory, // 👈 添加 checkpointer
});

// 使用（关键：thread_id）
async function chat(message: string, userId: string) {
  return await agent.invoke(
    { messages: [new HumanMessage(message)] },
    { configurable: { thread_id: userId } } // 👈 用户 ID
  );
}

// 测试
await chat('我喜欢吃披萨', 'user-1');
await chat('我喜欢什么？', 'user-1'); // "你喜欢吃披萨"
```

---

### 2. SqliteSaver（持久化存储）

```typescript
import { SqliteSaver } from '@langchain/langgraph-checkpoint-sqlite';
import Database from 'better-sqlite3';

// 创建 SQLite 数据库
const db = new Database(':memory:'); // 或 './checkpoints.db'
const checkpointer = SqliteSaver.fromDatabase(db);

const agent = createReactAgent({
  llm,
  tools,
  checkpointer, // 使用 SQLite 存储
});

// 即使重启程序，对话历史也不会丢失！
```

## 📊 完整示例：多用户聊天机器人

```typescript
import { MemorySaver } from '@langchain/langgraph';
import { createReactAgent } from '@langchain/langgraph/prebuilt';
import { ChatOpenAI } from '@langchain/openai';
import { HumanMessage } from '@langchain/core/messages';
import 'dotenv/config';

/**
 * 带记忆的聊天机器人
 * 功能：记住每个用户的对话历史
 */

class PersistentChatbot {
  private agent: any;

  constructor() {
    const memory = new MemorySaver();

    this.agent = createReactAgent({
      llm: new ChatOpenAI({ modelName: 'gpt-4o-mini', temperature: 0.7 }),
      tools: [],
      checkpointer: memory,
    });
  }

  async chat(userId: string, message: string): Promise<string> {
    const result = await this.agent.invoke(
      { messages: [new HumanMessage(message)] },
      { configurable: { thread_id: userId } }
    );

    const lastMessage = result.messages[result.messages.length - 1];
    return lastMessage.content as string;
  }

  // 获取用户的对话历史
  async getHistory(userId: string) {
    const state = await this.agent.getState({ configurable: { thread_id: userId } });
    return state.values.messages;
  }

  // 清除用户的对话历史
  async clearHistory(userId: string) {
    // 注意：MemorySaver 不支持删除，需要使用 SqliteSaver
    console.log(`清除用户 ${userId} 的历史（需要实现）`);
  }
}

// 测试
async function testMultiUser() {
  const bot = new PersistentChatbot();

  console.log('='.repeat(60));
  console.log('👤 用户 A 的对话');
  console.log('='.repeat(60));

  console.log('A:', await bot.chat('user-a', '我叫小明，我喜欢编程'));
  console.log('A:', await bot.chat('user-a', '我叫什么名字？'));

  console.log('\n' + '='.repeat(60));
  console.log('👤 用户 B 的对话');
  console.log('='.repeat(60));

  console.log('B:', await bot.chat('user-b', '我叫小红，我喜欢画画'));
  console.log('B:', await bot.chat('user-b', '我叫什么名字？'));

  console.log('\n' + '='.repeat(60));
  console.log('👤 回到用户 A');
  console.log('='.repeat(60));

  console.log('A:', await bot.chat('user-a', '我的爱好是什么？'));

  // 查看用户 A 的完整历史
  console.log('\n📋 用户 A 的对话历史:');
  const history = await bot.getHistory('user-a');
  history.forEach((msg, i) => {
    console.log(`${i + 1}. [${msg._getType()}]: ${msg.content}`);
  });
}

testMultiUser();
```

## 🎯 实战：带记忆的任务助手

创建 `src/21-task-assistant.ts`：

```typescript
import { Annotation, StateGraph } from '@langchain/langgraph';
import { MemorySaver } from '@langchain/langgraph';
import { ChatOpenAI } from '@langchain/openai';
import { HumanMessage, AIMessage, BaseMessage } from '@langchain/core/messages';
import 'dotenv/config';

/**
 * 任务管理助手（带记忆）
 * 功能：添加任务、查看任务、完成任务
 */

// State 定义
const TaskState = Annotation.Root({
  messages: Annotation<BaseMessage[]>({
    reducer: (x, y) => x.concat(y),
    default: () => [],
  }),
  tasks: Annotation<Array<{ id: number; title: string; done: boolean }>>({
    reducer: (_, y) => y, // 覆盖
    default: () => [],
  }),
  nextTaskId: Annotation<number>({
    reducer: (_, y) => y,
    default: () => 1,
  }),
});

type State = typeof TaskState.State;

const llm = new ChatOpenAI({ modelName: 'gpt-4o-mini', temperature: 0 });

// Agent 节点
async function agentNode(state: State): Promise<Partial<State>> {
  const userMessage = state.messages[state.messages.length - 1];
  const userInput = userMessage.content as string;

  // 简单的意图识别
  if (userInput.includes('添加') || userInput.includes('新建')) {
    const title = userInput.replace(/添加|新建|任务/g, '').trim();
    const newTask = { id: state.nextTaskId, title, done: false };

    return {
      tasks: [...state.tasks, newTask],
      nextTaskId: state.nextTaskId + 1,
      messages: [new AIMessage(`✅ 已添加任务: ${title}`)],
    };
  }

  if (userInput.includes('列表') || userInput.includes('查看')) {
    const taskList = state.tasks.map(t => `${t.done ? '✅' : '⬜'} ${t.id}. ${t.title}`).join('\n');

    return {
      messages: [new AIMessage(taskList || '暂无任务')],
    };
  }

  if (userInput.includes('完成')) {
    const idMatch = userInput.match(/\d+/);
    if (idMatch) {
      const taskId = parseInt(idMatch[0]);
      const tasks = state.tasks.map(t => (t.id === taskId ? { ...t, done: true } : t));

      return {
        tasks,
        messages: [new AIMessage(`✅ 任务 ${taskId} 已完成`)],
      };
    }
  }

  return {
    messages: [new AIMessage('请说：添加任务、查看列表、完成任务')],
  };
}

// 构建图
const memory = new MemorySaver();

const graph = new StateGraph(TaskState)
  .addNode('agent', agentNode)
  .addEdge('__start__', 'agent')
  .addEdge('agent', '__end__')
  .compile({ checkpointer: memory });

// 测试
async function testTaskAssistant() {
  const userId = 'user-task-123';

  async function chat(message: string) {
    const result = await graph.invoke(
      { messages: [new HumanMessage(message)] },
      { configurable: { thread_id: userId } }
    );

    const lastMsg = result.messages[result.messages.length - 1];
    console.log('🤖:', lastMsg.content);
    return result;
  }

  console.log('👤: 添加任务：学习 LangGraph');
  await chat('添加任务：学习 LangGraph');

  console.log('\n👤: 添加任务：写代码');
  await chat('添加任务：写代码');

  console.log('\n👤: 查看任务列表');
  await chat('查看任务列表');

  console.log('\n👤: 完成任务 1');
  await chat('完成任务 1');

  console.log('\n👤: 查看任务列表');
  const result = await chat('查看任务列表');

  // 查看完整 State
  console.log('\n📊 当前 State:');
  console.log('任务列表:', result.tasks);
}

testTaskAssistant();
```

## 💡 常见问题 FAQ

**Q: MemorySaver 和 SqliteSaver 的区别？**
A:

- MemorySaver：内存存储，重启后丢失
- SqliteSaver：文件存储，永久保存

**Q: 如何清除用户历史？**
A: 需要使用 SqliteSaver，然后删除对应的 thread_id 记录。

**Q: thread_id 必须是字符串吗？**
A: 是的，thread_id 必须是字符串类型。
