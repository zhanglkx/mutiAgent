## 📝 条件路由的本质

### 什么是条件路由?

**前端类比**：

```typescript
// React Router：根据路径跳转
<Route path="/home" element={<Home />} />
<Route path="/about" element={<About />} />

// LangGraph：根据 State 跳转
function router(state: State): string {
  if (state.needsHelp) return 'help';
  if (state.taskDone) return 'finish';
  return 'continue';
}
```

条件路由让 Agent 能够：
1. **自主决策**：根据当前状态选择下一步
2. **动态流转**：不同输入走不同路径
3. **循环执行**：实现 ReAct 模式

## 🛠️ 路由模式

### 1. 简单二分支路由

```typescript
import { Annotation, StateGraph } from '@langchain/langgraph';

const State = Annotation.Root({
  value: Annotation<number>({
    default: () => 0,
  }),
});

type StateType = typeof State.State;

// 路由函数：奇数或偶数
function oddOrEven(state: StateType): 'odd' | 'even' {
  return state.value % 2 === 0 ? 'even' : 'odd';
}

const graph = new StateGraph(State)
  .addNode('check', (s) => s)
  .addNode('odd_handler', (s) => {
    console.log('处理奇数:', s.value);
    return {};
  })
  .addNode('even_handler', (s) => {
    console.log('处理偶数:', s.value);
    return {};
  })

  .addEdge('__start__', 'check')

  // 条件路由
  .addConditionalEdges('check', oddOrEven, {
    odd: 'odd_handler',
    even: 'even_handler',
  })

  .addEdge('odd_handler', '__end__')
  .addEdge('even_handler', '__end__')

  .compile();

// 测试
await graph.invoke({ value: 5 });  // 输出：处理奇数: 5
await graph.invoke({ value: 8 });  // 输出：处理偶数: 8
```

### 2. 多分支路由

```typescript
type Priority = 'high' | 'medium' | 'low';

const TaskState = Annotation.Root({
  priority: Annotation<Priority>({
    default: () => 'medium',
  }),
  task: Annotation<string>({
    default: () => '',
  }),
});

type State = typeof TaskState.State;

// 根据优先级路由
function priorityRouter(state: State): string {
  switch (state.priority) {
    case 'high':
      return 'urgent';
    case 'medium':
      return 'normal';
    case 'low':
      return 'backlog';
    default:
      return 'normal';
  }
}

const graph = new StateGraph(TaskState)
  .addNode('router', (s) => s)
  .addNode('urgent', (s) => {
    console.log('🔥 紧急任务:', s.task);
    return {};
  })
  .addNode('normal', (s) => {
    console.log('📌 普通任务:', s.task);
    return {};
  })
  .addNode('backlog', (s) => {
    console.log('📋 待办任务:', s.task);
    return {};
  })

  .addEdge('__start__', 'router')

  .addConditionalEdges('router', priorityRouter, {
    urgent: 'urgent',
    normal: 'normal',
    backlog: 'backlog',
  })

  .addEdge('urgent', '__end__')
  .addEdge('normal', '__end__')
  .addEdge('backlog', '__end__')

  .compile();
```

### 3. 循环路由（ReAct 模式）

```typescript
const AgentState = Annotation.Root({
  messages: Annotation<BaseMessage[]>({
    reducer: (x, y) => x.concat(y),
    default: () => [],
  }),
  loopCount: Annotation<number>({
    reducer: (x, y) => x + y,
    default: () => 0,
  }),
});

type State = typeof AgentState.State;

// 决定是否继续循环
function shouldContinue(state: State): 'continue' | 'end' {
  const lastMessage = state.messages[state.messages.length - 1];

  // 检查是否需要调用工具
  if ('tool_calls' in lastMessage && lastMessage.tool_calls?.length > 0) {
    return 'continue';  // 有工具调用，继续
  }

  // 检查是否达到最大循环次数
  if (state.loopCount >= 10) {
    return 'end';  // 防止无限循环
  }

  return 'end';  // 没有工具调用，结束
}

const graph = new StateGraph(AgentState)
  .addNode('agent', agentNode)
  .addNode('tools', toolsNode)

  .addEdge('__start__', 'agent')

  // 核心：循环路由
  .addConditionalEdges('agent', shouldContinue, {
    continue: 'tools',  // 执行工具
    end: '__end__',     // 结束
  })

  .addEdge('tools', 'agent')  // 工具执行后回到 agent

  .compile();
```

**可视化**：

```
        ┌──────────┐
START → │  agent   │
        └────┬─────┘
             ↓
        shouldContinue()
         ╱        ╲
   continue       end
      ╱              ╲
     ↓                ↓
┌─────────┐         END
│  tools  │
└────┬────┘
     ↓
  [回到 agent]
```

## 📊 完整示例：智能客服 Agent

创建 `src/15-customer-service-agent.ts`：

```typescript
import { Annotation, StateGraph } from '@langchain/langgraph';
import { ChatOpenAI } from '@langchain/openai';
import { HumanMessage, AIMessage, BaseMessage, SystemMessage } from '@langchain/core/messages';
import 'dotenv/config';

/**
 * 智能客服 Agent
 * 功能：根据用户意图分类，路由到不同处理流程
 */

// 定义意图类型
type Intent = 'query_order' | 'refund' | 'complaint' | 'general';

// State 定义
const CustomerServiceState = Annotation.Root({
  messages: Annotation<BaseMessage[]>({
    reducer: (x, y) => x.concat(y),
    default: () => [],
  }),
  intent: Annotation<Intent | null>({
    reducer: (_, y) => y,
    default: () => null,
  }),
  orderId: Annotation<string | null>({
    reducer: (_, y) => y,
    default: () => null,
  }),
});

type State = typeof CustomerServiceState.State;

const llm = new ChatOpenAI({ modelName: 'gpt-4o-mini', temperature: 0 });

// 节点1：识别意图
async function classifyIntentNode(state: State): Promise<Partial<State>> {
  const userMessage = state.messages[state.messages.length - 1];

  const response = await llm.invoke([
    new SystemMessage(`你是意图识别专家。
请识别用户意图，从以下选项中选择一个（只回复选项，不要其他内容）：
- query_order: 查询订单
- refund: 退款
- complaint: 投诉
- general: 一般咨询`),
    userMessage,
  ]);

  const intent = (response.content as string).trim() as Intent;
  console.log(`🎯 识别意图: ${intent}`);

  return { intent };
}

// 节点2：查询订单
async function handleQueryOrderNode(state: State): Promise<Partial<State>> {
  console.log('📦 处理订单查询...');

  // 提取订单号
  const userMessage = state.messages[state.messages.length - 1];
  const orderIdMatch = (userMessage.content as string).match(/\d{6,}/);
  const orderId = orderIdMatch ? orderIdMatch[0] : null;

  if (!orderId) {
    return {
      messages: [
        new AIMessage('请提供您的订单号（6位数字）'),
      ],
    };
  }

  // 模拟查询订单
  const orderInfo = {
    id: orderId,
    status: '配送中',
    expectedDelivery: '明天下午',
  };

  return {
    orderId,
    messages: [
      new AIMessage(
        `您的订单 ${orderId} 当前状态：${orderInfo.status}，预计${orderInfo.expectedDelivery}送达。`
      ),
    ],
  };
}

// 节点3：处理退款
async function handleRefundNode(state: State): Promise<Partial<State>> {
  console.log('💰 处理退款请求...');

  return {
    messages: [
      new AIMessage(
        '已为您提交退款申请，预计3-5个工作日到账。如需帮助请联系人工客服。'
      ),
    ],
  };
}

// 节点4：处理投诉
async function handleComplaintNode(state: State): Promise<Partial<State>> {
  console.log('⚠️ 处理投诉...');

  return {
    messages: [
      new AIMessage(
        '非常抱歉给您带来不便。您的投诉已记录，我们会在24小时内回复您。投诉编号：CP' +
          Date.now()
      ),
    ],
  };
}

// 节点5：一般咨询
async function handleGeneralNode(state: State): Promise<Partial<State>> {
  console.log('💬 处理一般咨询...');

  const userMessage = state.messages[state.messages.length - 1];
  const response = await llm.invoke([
    new SystemMessage('你是友好的客服，简洁回答用户问题。'),
    userMessage,
  ]);

  return {
    messages: [new AIMessage(response.content as string)],
  };
}

// 路由函数
function intentRouter(state: State): string {
  switch (state.intent) {
    case 'query_order':
      return 'query_order';
    case 'refund':
      return 'refund';
    case 'complaint':
      return 'complaint';
    case 'general':
      return 'general';
    default:
      return 'general';
  }
}

// 构建图
const graph = new StateGraph(CustomerServiceState)
  .addNode('classify', classifyIntentNode)
  .addNode('query_order', handleQueryOrderNode)
  .addNode('refund', handleRefundNode)
  .addNode('complaint', handleComplaintNode)
  .addNode('general', handleGeneralNode)

  .addEdge('__start__', 'classify')

  // 根据意图路由
  .addConditionalEdges('classify', intentRouter, {
    query_order: 'query_order',
    refund: 'refund',
    complaint: 'complaint',
    general: 'general',
  })

  .addEdge('query_order', '__end__')
  .addEdge('refund', '__end__')
  .addEdge('complaint', '__end__')
  .addEdge('general', '__end__')

  .compile();

// 测试
async function testCustomerService() {
  const testCases = [
    '我想查询订单 123456 的物流',
    '我要退款，商品有质量问题',
    '你们的客服态度太差了，我要投诉',
    '请问你们的营业时间是几点？',
  ];

  for (const query of testCases) {
    console.log('\n' + '='.repeat(60));
    console.log('👤 用户:', query);
    console.log('='.repeat(60));

    const result = await graph.invoke({
      messages: [new HumanMessage(query)],
    });

    const lastMessage = result.messages[result.messages.length - 1];
    console.log('🤖 客服:', lastMessage.content);
  }
}

testCustomerService();
```

## 🔍 高级路由技巧

### 1. AI 驱动的路由

```typescript
// 让 AI 决定下一步
async function aiRouter(state: State): Promise<string> {
  const response = await llm.invoke([
    new SystemMessage(`根据对话历史，决定下一步操作。
选项：
- need_more_info: 需要更多信息
- ready_to_act: 可以执行操作
- task_complete: 任务完成

只回复选项名称。`),
    ...state.messages,
  ]);

  return (response.content as string).trim();
}

// 注意：AI 路由函数必须是 async，需要特殊处理
// LangGraph 的路由函数必须同步，所以我们在节点中做决策

async function decisionNode(state: State): Promise<Partial<State>> {
  const decision = await aiRouter(state);
  return { nextAction: decision };
}

function router(state: State): string {
  return state.nextAction || 'default';
}
```

### 2. 基于概率的路由

```typescript
function randomRouter(state: State): 'path_a' | 'path_b' {
  // 80% 走 path_a, 20% 走 path_b
  return Math.random() < 0.8 ? 'path_a' : 'path_b';
}
```

### 3. 基于时间的路由

```typescript
function timeBasedRouter(state: State): string {
  const hour = new Date().getHours();

  if (hour >= 9 && hour < 18) {
    return 'business_hours';  // 工作时间
  }
  return 'after_hours';  // 非工作时间
}
```

## 💡 常见问题 FAQ

**Q: 路由函数可以访问外部变量吗？**
A: 可以，但不推荐。最好把所有信息都放在 State 中。

**Q: 如何处理路由函数返回未定义的节点？**
A: 图会抛出错误。确保路由函数返回的节点名都在映射表中定义。

**Q: 可以动态修改路由逻辑吗？**
A: 不可以。路由逻辑在图编译时就固定了。如果需要动态逻辑，把它放在 State 中。

**Q: 路由函数可以修改 State 吗？**
A: 不可以！路由函数只能读 State，不能写。修改 State 要在节点函数中完成。
